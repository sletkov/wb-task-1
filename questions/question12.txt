12.	Что выведет данная программа и почему?


func main() {
  n := 0
  if true {
    //внутри данного условного оператора существует свой scope(область видимости)
     n := 1 //переменная создалась в scope условного оператора
     n++ //инкремент произойдет с переменной в scope условного оператора
  }
  fmt.Println(n) //Программа выведет переменную n своего scope
}

Вывод программы:
0
