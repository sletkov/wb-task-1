14.	Что выведет данная программа и почему?

func main() {
  slice := []string{"a", "a"}

  func(slice []string) {
     slice = append(slice, "a") // переменной slice присвоили новый слайс, созданный функцией append путем слияния старого slice и элемента "a"
     // slice теперь выглядит так ["a","a","a"]
     slice[0] = "b" // теперь slice выглядит так ["b","a","a"]
     slice[1] = "b" // теперь slice выглядит так ["b","b","a"]
     fmt.Print(slice) // вывод будет выглядеть так ["b","b","a"]
  }(slice) //передали слайс в качестве аргумента
  fmt.Print(slice) //здесь работаем с нашим старым слайсом и выводим [a a]
}

Вывод программы
[b b a] [a a]
