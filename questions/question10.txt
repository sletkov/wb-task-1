10.	Что выведет данная программа и почему?

func update(p *int) {
  b := 2
  p = &b
}

func main() {
  var (
     a = 1
     p = &a
  )
  fmt.Println(*p) // Программа выведет 1, тк в p лежит указатель на a
  update(p) // Вызов функции update не повлияет на p, так как внутри функции update создается своя копия переменной p
  fmt.Println(*p) // В p все еще лежит указатель на a. Программа выведет 1
}

Программа выведет:
1
1
