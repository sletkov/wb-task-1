13.	Что выведет данная программа и почему?


func someAction(v []int8, b int8) {
  v[0] = 100 //изменяем значение первого элемента слайса. тк слайс хранит ссылку на массив, то все изменяется
  v = append(v, b) //v присваивается новый slice, созданный функцией append, ктр состоит из старого слайса v и элемента b
  // v = [100,2,3,4,5,6], но это уже другое v и другой slice
}

func main() {
  var a = []int8{1, 2, 3, 4, 5}
  someAction(a, 6)// функция изменяет только первый элемент массива
  fmt.Println(a)
}

Программа выведет
[100 2 3 4 5]
